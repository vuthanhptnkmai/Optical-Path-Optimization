76template<typename T, typename U>
77std::pair<int, int> Detector<T, U>::mapToPixelGrid(const vec3<T>& intersectionPoint) const {
78 // Get required vectors and dimensions
79 auto planarRect = dynamic_cast<PlanarRectangle<T, U>*>(this->surface.get());
80 if (!planarRect) throw std::runtime_error("Surface is not a PlanarRectangle.");
81
82 vec3<T> D_height = planarRect->getHeightDirection();
83 vec3<T> D_width = planarRect->getNormal().cross(D_height).normalized();
84
85 vec3<T> center = planarRect->getPosition();
86 T width = planarRect->getWidth();
87 T height = planarRect->getHeight();
88
89 T u = (intersectionPoint - center).dot(D_width);
90 T v = (intersectionPoint - center).dot(D_height);
91
92 // Normalize local coordinates to [0, 1]
93 T u_norm = (u + width / 2.0) / width;
94 T v_norm = (v + height / 2.0) / height;
95
96 // Map to pixel grid coordinates
97 int xPixel = static_cast<int>(u_norm * pixelWidth);
98 int yPixel = static_cast<int>(v_norm * pixelHeight);
99
100 return std::make_pair(xPixel, yPixel);
101}
