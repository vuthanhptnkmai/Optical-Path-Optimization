47void Detector<T, U>::handleLight(Ray<T, U>& ray, const vec3<T>& intersectionPoint) {
48 // Assuming a planar detector for simplicity
49 vec3<T> localCoords = intersectionPoint - this->getPosition();
50 vec3<T> normal = this->getNormal().normalized();
51
52 // Project the intersection point onto the detector's plane
53 T x = localCoords.dot(normal.cross(vec3<T>(0,1,0)));
54 T y = localCoords.dot(normal.cross(vec3<T>(1,0,0)));
55
56 // Normalize these coordinates to the range [0,1]
57 x = (x / this->surface->getWidth()) + 0.5;
58 y = (y / this->surface->getHeight()) + 0.5;
59
60 // Convert normalized coordinates to pixel indices
61 int pixelX = std::min(std::max(int(x * pixelWidth), 0), pixelWidth - 1);
62 int pixelY = std::min(std::max(int(y * pixelHeight), 0), pixelHeight - 1);
63
64 // Increment the intensity at the pixel
65 // Assuming the intensity is 1 for each ray for simplicity
66 pixelData[pixelX][pixelY] += 1;
67}
