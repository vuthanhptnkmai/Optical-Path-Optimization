34 void saveImageAsPNG(const std::string& filename, const Eigen::MatrixXf& pixelGrid) {
35 int width = pixelGrid.cols(), height = pixelGrid.rows();
36 std::vector<uint8_t> image(width * height);
37
38 for (int y = 0; y < height; ++y) {
39 for (int x = 0; x < width; ++x) {
40 image[y * width + x] = static_cast<uint8_t>(pixelGrid(y, x) * 255);
41 }
42 }
43
44 stbi_write_png(filename.c_str(), width, height, 1, image.data(), width);
45 }
