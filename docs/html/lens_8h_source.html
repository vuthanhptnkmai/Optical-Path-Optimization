45template<typename T, typename U>
46void ThinLens<T, U>::refract_approx(Ray<T, U>& ray, const vec3<T>& intersectionPoint) {
47 vec3<T> O = ray.position;
48 vec3<T> D = ray.direction;
49 vec3<T> P = this->surface->getPosition();
50 vec3<T> N = -this->surface->getNormal();
51
52 vec3<T> PR = P - O; // principal ray from the light source to the center of the lens
53 T objectDistance = PR.dot(N); // distance from the object to the lens
54 T imageDistance = 1 / ((1 / focalLength) - (1 / objectDistance)); // distance to the image using the thin lens approximation formula
55
56 vec3<T> I = P + imageDistance * PR.normalized(); // calculate the image position
57
58 ray.position = intersectionPoint; // change the position of the ray to the intersection of the ray with the lens
59
60 if (imageDistance < 0) { // Virtual image scenario
61 ray.direction = (intersectionPoint - I).normalized(); // reverse the direction towards the virtual image
62 } else {
63 ray.direction = (I - intersectionPoint).normalized(); // change the direction of the ray to the normalized vector from that intersection point to the image
64 }
65}
